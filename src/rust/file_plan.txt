src/
├── lib.rs                    # Library entry point, re-exports main modules
├── main.rs                   # CLI entry point (if needed)
│

│   ├── vial.rs              # Vial struct, operations on vials

│   └── color.rs             # Color handling, reserved colors, color validation

│   ├── mod.rs               # Solver module exports

│   ├── safe_game_solver.rs  # Safe move detection
│   └── strategy.rs          # Solve methods (MIX, BFS, DFS, DFR)

├── analysis/

│   └── stats.rs             # Statistical analysis of solutions

├── io/

│   ├── formatter.rs         # Format game display, move display
│   └── file_io.rs           # File saving/loading operations
│
├── display/
│   ├── mod.rs               # Display module exports
│   ├── game_display.rs      # Display Game state (vials, moves)
│   ├── big_display.rs       # BigSolutionDisplay equivalent
│   └── colors.rs            # Color formatting/ANSI codes
│
├── types/

│   └── constants.rs         # Game constants (NUM_SPACES_PER_VIAL, etc)
│
└── utils/
    ├── mod.rs               # Utilities module exports
    └── helpers.rs           # Helper functions (fPercent, etc)
